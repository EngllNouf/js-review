Introduction to JavaScript

JavaScript is a versatile programming language that powers the modern web. It allows you to create interactive websites, build web applications, develop server-side programs, and even create mobile apps.

Fun Fact: Despite its name, JavaScript has no direct relationship with Java. The naming was primarily a marketing decision!

Client-Side Scripting

Run code in the browser to create dynamic web pages

Server-Side Development

Build backend applications with Node.js

Your First JavaScript Code

Operators

Operators perform operations on variables and values.

Common Operators 

// Arithmetic Operators
let sum = 5 + 3;       // Addition
let difference = 10 - 4; // Subtraction
let product = 6 * 2;    // Multiplication
let quotient = 15 / 3;  // Division
let remainder = 17 % 5; // Modulus

// Comparison Operators
let isEqual = 5 === 5;    // Strict equality
let isNotEqual = 5 !== 3; // Strict inequality
let isGreater = 10 > 5;   // Greater than
let isLess = 3 < 7;      // Less than

// Logical Operators
let and = true && false;  // Logical AND
let or = true || false;   // Logical OR
let not = !true;         // Logical NOT

Control Flow

Control flow statements determine the order in which code is executed based on certain conditions.

If Statements 

// Basic if statement
let age = 18;

if (age >= 18) {
    console.log("You are an adult");
} else {
    console.log("You are a minor");
}

// Multiple conditions
let score = 85;

if (score >= 90) {
    console.log("Grade: A");
} else if (score >= 80) {
    console.log("Grade: B");
} else if (score >= 70) {
    console.log("Grade: C");
} else {
    console.log("Grade: F");
}

Functions

Functions are reusable blocks of code that perform specific tasks.

Function Examples 

// Regular function
function greet(name) {
    return "Hello, " + name + "!";
}

// Arrow function
const add = (a, b) => a + b;

// Function with default parameters
function multiply(a, b = 1) {
    return a * b;
}

// Function expression
const square = function(x) {
    return x * x;
};

// Using the functions
console.log(greet("NAZ"));        // Hello, NAZ!
console.log(add(5, 3));          // 8
console.log(multiply(4));        // 4
console.log(square(5));         // 25

Arrays

Arrays are ordered lists of values that can store multiple items in a single variable.

Array Operations 

// Creating arrays
let fruits = ['apple', 'banana', 'orange'];
let numbers = [1, 2, 3, 4, 5];

// Accessing elements
console.log(fruits[0]);     // 'apple'
console.log(numbers[2]);    // 3

// Array methods
fruits.push('mango');      // Add to end
fruits.pop();              // Remove from end
fruits.unshift('grape');   // Add to start
fruits.shift();            // Remove from start

// Array operations
let length = fruits.length;    // Array length
let joined = fruits.join(', '); // Join elements
let includes = fruits.includes('banana'); // Check if exists
let found = fruits.find(fruit => fruit === 'apple'); // Find element

Objects

Objects are collections of related data and functionality stored in key-value pairs.

Working with Objects 

// Creating an object
let person = {
    name: 'John',
    age: 25,
    isStudent: true,
    hobbies: ['reading', 'music'],
    address: {
        city: 'New York',
        country: 'USA'
    },
    sayHello: function() {
        return `Hello, my name is ${this.name}!`;
    }
};

// Accessing properties
console.log(person.name);        // 'John'
console.log(person['age']);      // 25
console.log(person.hobbies[0]);  // 'reading'
console.log(person.sayHello());  // 'Hello, my name is John!'

// Modifying objects
person.age = 26;
person.email = 'john@example.com';  // Add new property
delete person.isStudent;            // Remove property

DOM Manipulation

The Document Object Model (DOM) allows JavaScript to interact with HTML elements on the page.

DOM Operations 

// Selecting elements
let element = document.getElementById('myId');
let elements = document.getElementsByClassName('myClass');
let query = document.querySelector('.myClass');

// Modifying elements
element.innerHTML = 'New content';
element.style.color = 'blue';
element.classList.add('active');
element.setAttribute('data-value', '123');

// Creating elements
let div = document.createElement('div');
div.textContent = 'New Element';
document.body.appendChild(div);

// Event handling
element.addEventListener('click', function(event) {
    console.log('Element clicked!');
});

Asynchronous JavaScript

Asynchronous programming allows code to run in the background without blocking the main execution thread.

Async Examples 

// Promises
let promise = new Promise((resolve, reject) => {
    setTimeout(() => {
        resolve('Data received!');
    }, 2000);
});

promise.then(data => console.log(data))
       .catch(error => console.error(error));

// Async/Await
async function fetchData() {
    try {
        let response = await fetch('https://api.example.com/data');
        let data = await response.json();
        return data;
    } catch (error) {
        console.error('Error:', error);
    }
}

// SetTimeout
setTimeout(() => {
    console.log('Executed after 2 seconds');
}, 2000);
